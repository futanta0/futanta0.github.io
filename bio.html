Data setup: treatment and placebo groups 
library(survival)

mp      <- data.frame(time  = c(6,6,6,7,10,13,16,22,23, 6,9,10,11,17,19,20,25,32,32,34,35),
                       event = c(rep(1, 9), rep(0, 12)))

placebo <- data.frame(time  = c(1,1,2,2,3,4,4,5,5,8,8,8,8,11,11,12,12,15,17,22,23),
                       event = rep(1, 21))
(a) KM estimates and base R plot 
km1 <- survfit(Surv(time, event) ~ 1, data = mp)
km2 <- survfit(Surv(time, event) ~ 1, data = placebo)

summary(km1)
summary(km2)

plot(km1, col = "red", lwd = 2, xlab = "Time (weeks)", ylab = "Survival Probability",
     main = "Kaplan-Meier Curves", mark.time = TRUE, conf.int = FALSE)

legend("topright", legend = c("Treatment", "Placebo"), col = c("red", "blue"), lwd = 2)
(a) ggplot2/survminer alternative 
library(survminer)

km_fit_gg <- survfit(Surv(time, event) ~ group,
                      data = rbind(transform(mp, group = "Treatment"),
                                   transform(placebo, group = "Placebo")))

           palette = c("blue", "red"),      # alphabetical group order: Placebo, Treatment
           conf.int = FALSE,
           legend.title = "",
           legend.labs = c("Placebo", "Treatment"),

HYPERT — Hypertension Screening (Logistic Regression)

Import HYPERT.csv, sample 650, export to SPSS 
HYPERT <- read.csv("HYPERT.csv", header = TRUE)
str(HYPERT)

set.seed(123456)                          # replace 123456 with your own ID number
HYPERT2 <- HYPERT[sample(nrow(HYPERT), size = 650, replace = FALSE), ]
nrow(HYPERT2)

library(haven)
write_sav(HYPERT2, "HYPERT.sav")
(a) Missing observations 
sum(is.na(HYPERT2))
colSums(is.na(HYPERT2))
(b) Calculate BMI and BMICLASS 
# Calculate BMI
HYPERT2$BMI <- HYPERT2$weight / (HYPERT2$height/100)^2

# Create BMI class
HYPERT2$BMICLASS <- cut(HYPERT2$BMI,
                         breaks = c(-Inf, 20, 25, 30, Inf),
                         labels = c(1, 2, 3, 4),
                         right = FALSE)

# Check the result
table(HYPERT2$BMICLASS)
(c) Risk of ISH for women + 95% CI 
library(DescTools)

women <- subset(HYPERT2, sex == 1)
x <- sum(women$ISH == 1); n <- nrow(women)

BinomCI(x, n, method = "wald")
(d) Association between ISH and BMICLASS 
tab <- table(HYPERT2$BMICLASS, HYPERT2$ISH)
tab

prop.table(tab, margin = 1)

chisq.test(tab)
(e) Logistic regression model + odds ratios 
HYPERT2$sex      <- factor(HYPERT2$sex, labels = c("Male", "Female"))
HYPERT2$smoke93  <- factor(HYPERT2$smoke93, labels = c("No", "Yes"))

model <- glm(ISH ~ age + sex + smoke93 + BMICLASS, data = HYPERT2, family = binomial(link = "logit"))
summary(model)

# Odds ratios and 95% CIs
exp(coef(model))
exp(confint(model))

se <- sqrt(p * (1 - p) / n)

c(p = p, n = n, se = se, lower = p - 1.96 * se, upper = p + 1.96 * se)

t

c(null_deviance = model$null.deviance,
  residual_deviance = model$deviance,
  AIC = AIC(model))
(k) Classification performance (confusion matrix) 
predicted_prob  <- predict(model, type = "response")
